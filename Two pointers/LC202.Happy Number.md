```
class Solution:
    def isHappy(self, n: int) -> bool:
        def caculate(n: int) -> int:
            temp = 0
            while n > 0:
                temp += (n % 10) ** 2
                n //= 10
            return temp

        slow = n
        fast = caculate(n)

        while fast != 1 and slow != fast:
            slow = caculate(slow)
            fast = caculate(caculate(fast))  # 快指针走两步

        return fast == 1
```
